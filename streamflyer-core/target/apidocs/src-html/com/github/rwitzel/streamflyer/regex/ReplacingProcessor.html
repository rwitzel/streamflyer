017package com.github.rwitzel.streamflyer.regex;
018
019import java.util.ArrayList;
020import java.util.List;
021import java.util.regex.MatchResult;
022import java.util.regex.Matcher;
023
024import com.github.rwitzel.streamflyer.internal.thirdparty.ZzzValidate;
025
026/**
027 * This {@link MatchProcessor} replaces the matching text (given by the {@link MatchResult}) with the configured
028 * replacement.
029 * <p>
030 * The replacement can be defined with the same syntax as defined by
031 * {@link Matcher#appendReplacement(StringBuffer, String)}
032 * <p>
033 * EXAMPLES: <code><pre class="prettyprint lang-java">
034new ReplacingProcessor("aaa") // replaces each match with "aaa"
035new ReplacingProcessor("$2") // replaces each match with the content of the second {@link MatchResult#group() group} of the match
036new ReplacingProcessor("aaa$2bbb$1ccc") // replaces each match with "aaa$2bbb$1ccc"
037                           // wherein $2 and $1 are replaced with the content
038                           // of the second and the first group of the match
039</pre></code>
040 * 
041 * @author rwoo
042 * @since 18.06.2011
043 */
044public class ReplacingProcessor extends AbstractMatchProcessor implements MatchProcessor {
045
046    //
047    // injected properties
048    //
049
050    /**
051     * This list contains Strings and Numbers. The latter ones refer to groups.
052     */
053    private List<Object> parts = null;
054
055    /**
056     * If not null, then the replacement do not uses groups. Then this property contains the entire replacement.
057     */
058    private CharSequence replacementWithoutGroupReferences;
059
060    //
061    // constructors
062    //
063
064    /**
065     * For unit tests only.
066     */
067    public ReplacingProcessor() {
068        // nothing to do here
069    }
070
071    /**
072     * @param replacement
073     */
074    public ReplacingProcessor(String replacement) {
075        super();
076
077        // validate arguments
078        ZzzValidate.notNull(replacement, "replacement must not be null");
079
080        // parse replacement
081        this.parts = parseReplacement(replacement);
082
083        if (parts.size() == 1 && (parts.get(0) instanceof CharSequence)) {
084
085            this.replacementWithoutGroupReferences = (CharSequence) parts.get(0);
086
087        } else {
088
089            this.replacementWithoutGroupReferences = null;
090
091        }
092    }
093
094    /**
095     * Package-private for JUnit-tests.
096     * 
097     * @return Returns the parts of the matchProcessor
098     */
099    List<Object> parseReplacement(String replacement) {
100
101        List<Object> compiledReplacement = new ArrayList<Object>();
102
103        // we look for escaped literals and references to groups
104        int position = 0;
105        StringBuilder notGroupReference = new StringBuilder();
106
107        while (position < replacement.length()) {
108            char ch = replacement.charAt(position);
109
110            // reference to group?
111            if (ch == '$') {
112
113                if (notGroupReference.length() != 0) {
114                    compiledReplacement.add(notGroupReference.toString());
115                    notGroupReference.setLength(0);
116                }
117
118                position++;
119
120                int groupNumberStartPosition = position;
121                if (groupNumberStartPosition == replacement.length()) {
122                    throw new IllegalArgumentException("group reference $ " + "without number at the end " + "of the "
123                            + "replacement string (" + replacement + ")");
124                }
125
126                ch = replacement.charAt(position);
127
128                while ('0' <= ch && ch <= '9') {
129                    position++;
130                    if (position == replacement.length()) {
131                        break;
132                    }
133                    ch = replacement.charAt(position);
134                }
135                // position is the position of the first character that does not
136                // belong to the group reference
137
138                if (position == groupNumberStartPosition) {
139                    throw new IllegalArgumentException("invalid reference to group at position " + position
140                            + " in replacement string " + replacement.substring(0, position) + "[]"
141                            + replacement.substring(position));
142                }
143
144                String groupNumberString = replacement.substring(groupNumberStartPosition, position);
145                int groupNumber = Integer.parseInt(groupNumberString);
146
147                compiledReplacement.add(groupNumber);
148            }
149            // escaped literal?
150            else if (ch == '\\') {
151
152                // skip the backslash
153                position++;
154                ch = replacement.charAt(position);
155
156                notGroupReference.append(ch);
157                position++;
158            }
159            // this is a normal character
160            else {
161
162                notGroupReference.append(ch);
163                position++;
164            }
165        }
166
167        if (notGroupReference.length() != 0) {
168            compiledReplacement.add(notGroupReference.toString());
169        }
170
171        return compiledReplacement;
172    }
173
174    /**
175     * Replaces the match as given by the {@link MatchResult} with the configured replacement.
176     * 
177     * @see com.github.rwitzel.streamflyer.regex.MatchProcessor#process(java.lang.StringBuilder, int,
178     *      java.util.regex.MatchResult)
179     */
180    @Override
181    public MatchProcessorResult process(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
182            MatchResult matchResult) {
183
184        int start = matchResult.start();
185        int end = matchResult.end();
186
187        if (replacementWithoutGroupReferences != null) {
188
189            characterBuffer.delete(start, end);
190            characterBuffer.insert(start, replacementWithoutGroupReferences);
191            return createResult(matchResult, start + replacementWithoutGroupReferences.length(), true);
192        } else {
193
194            // (1) create replacement
195            // (one could set a reasonable value for the initial size of the
196            // string builder somewhere)
197            StringBuilder replacement = new StringBuilder();
198
199            for (Object part : parts) {
200                if (part instanceof Integer) {
201                    // append the value of the referred group
202                    replacement.append(matchResult.group((Integer) part));
203                } else {
204                    // append the static part
205                    replacement.append(part); // part is instance of String
206                }
207            }
208
209            // (2) replace the match with the replacement
210            characterBuffer.delete(start, end);
211            characterBuffer.insert(start, replacement);
212
213            return createResult(matchResult, start + replacement.length(), true);
214        }
215
216    }
217}
