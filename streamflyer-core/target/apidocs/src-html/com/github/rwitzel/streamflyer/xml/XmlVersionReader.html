017package com.github.rwitzel.streamflyer.xml;
018
019import java.io.BufferedReader;
020import java.io.IOException;
021import java.io.Reader;
022import java.util.regex.Matcher;
023import java.util.regex.Pattern;
024
025/**
026 * This reader makes the XML version of the XML document in the character stream available.
027 * 
028 * @author rwoo
029 * @since 27.06.2011
030 */
031public class XmlVersionReader extends BufferedReader {
032
033    private String xmlVersion;
034
035    //
036    // constructors
037    //
038
039    public XmlVersionReader(Reader in) throws IOException {
040        super(in);
041
042        String prolog = prolog();
043        xmlVersion = xmlVersion(prolog);
044    }
045
046    //
047    // private methods
048    //
049
050    private String xmlVersion(String prolog) {
051
052        // (Should we do aware of BOMs here? No. I consider it the
053        // responsibility of the caller to provide characters without BOM.)
054
055        Matcher matcher = Pattern.compile("<\\?xml[^>]*version\\s*=\\s*['\"]((1.0)|(1.1))['\"].*").matcher(prolog);
056        if (matcher.matches()) {
057            return matcher.group(1);
058        } else {
059            // the default, see class comment.
060            return "1.0";
061        }
062    }
063
064    private String prolog() throws IOException {
065
066        // the prolog may contain many, many whitespace characters -> 4096
067        final int MAX_SNIFFED_CHARACTERS = 4096;
068
069        mark(MAX_SNIFFED_CHARACTERS);
070        try {
071
072            char[] cbuf = new char[MAX_SNIFFED_CHARACTERS];
073            int read = read(cbuf);
074            if (read == -1) {
075                return "";
076            } else {
077                return new String(cbuf, 0, read);
078            }
079        } finally {
080            reset();
081        }
082    }
083
084    //
085    // public methods
086    //
087
088    /**
089     * @return Returns the XML version read from the character stream. If there is XML prolog given, then version "1.0"
090     *         is assumed and returned.
091     */
092    public String getXmlVersion() {
093        return xmlVersion;
094    }
095}
