016package com.github.rwitzel.streamflyer.regex.addons.stateful;
017
018import java.util.regex.MatchResult;
019
020import com.github.rwitzel.streamflyer.regex.MatchProcessor;
021import com.github.rwitzel.streamflyer.regex.MatchProcessorResult;
022import com.github.rwitzel.streamflyer.regex.OnStreamMatcher;
023import com.github.rwitzel.streamflyer.regex.RegexModifier;
024import com.github.rwitzel.streamflyer.regex.addons.util.DelegatingMatcher;
025
026/**
027 * This {@link MatchProcessor} changes the {@link OnStreamMatcher} that is used by a {@link RegexModifier} if another
028 * {@link State} is reached.
029 * 
030 * @author rwoo
031 * @since 1.1.0
032 */
033public class StateMachine implements MatchProcessor {
034
035    /**
036     * The most recently reached state.
037     */
038    private State currentState;
039
040    /**
041     * The transitions that starts from {@link #currentState}.
042     */
043    private Transitions transitions;
044
045    /**
046     * The matcher that is used by the {@link RegexModifier}. By changing the delegate we can exchange the regex the
047     * modifier looks for. The matcher looks for the tokens that belong to the end states of {@link #transitions}.
048     */
049    private DelegatingMatcher delegatingMatcher;
050
051    /**
052     * 
053     * @param initialState
054     *            the initial state for the state machine
055     * @param delegatingMatcher
056     *            See {@link #delegatingMatcher}.
057     */
058    public StateMachine(State initialState, DelegatingMatcher delegatingMatcher) {
059        super();
060        this.delegatingMatcher = delegatingMatcher;
061        changeStateTo(initialState);
062    }
063
064    /**
065     * @return the currentState
066     */
067    public State getCurrentState() {
068        return currentState;
069    }
070
071    /**
072     * Changes the {@link #currentState} to the given state.
073     * 
074     * @param state
075     */
076    protected void changeStateTo(State state) {
077        currentState = state;
078        transitions = state.getTransitions();
079        delegatingMatcher.setDelegate(state.getTransitions().getMatcher());
080    }
081
082    @Override
083    public MatchProcessorResult process(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
084            MatchResult matchResult) {
085
086        // modify stream
087        MatchProcessorResult result = transitions.process(characterBuffer, firstModifiableCharacterInBuffer,
088                matchResult);
089
090        // change state if needed
091        State newState = transitions.pollNewState();
092        if (newState != currentState) {
093            changeStateTo(newState);
094        }
095
096        return result;
097    }
098
099}
