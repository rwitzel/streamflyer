016package com.github.rwitzel.streamflyer.regex.addons.util;
017
018import java.util.regex.Pattern;
019
020/**
021 * This class supports the creation of regular expressions with embedded flag expressions.
022 * 
023 * @author rwoo
024 * @since 1.1.0
025 */
026public class EmbeddedFlagUtil {
027
028    /**
029     * Creates a regular expression with an embedded flag expression.
030     * <p>
031     * Supports all flags of JDK7 {@link java.util.regex.Pattern}, i.e. the following flags:
032     * <ul>
033     * <li>{@link Pattern#CASE_INSENSITIVE}
034     * <li>{@link Pattern#UNIX_LINES}
035     * <li>{@link Pattern#MULTILINE}
036     * <li>{@link Pattern#DOTALL}
037     * <li>{@link Pattern#UNICODE_CASE}
038     * <li>{@link Pattern#COMMENTS}
039     * </ul>
040     * <p>
041     * EXAMPLE:
042     * <ul>
043     * <li>For <code>("abc", Pattern.CASE_INSENSITIVE ^ Pattern.MULTILINE)</code> the method returns <code>
044     *         "(?im:abc)"</code>.</li>
045     * <li>For <code>("abc", 0)</code> the method returns <code>"abc"</code>.</li>
046     * </ul>
047     * 
048     * @param regex
049     * @param flags
050     * @return Returns the given regex enriched with an embedded flag expression that represents the given flags. If
051     *         there is no flag given, the returned regex is equal to the given regex.
052     */
053    public String embedFlags(String regex, int flags) {
054        if (flags == 0) {
055            return regex;
056        } else {
057            return "(?" + mapFlagsToEmbeddedFlags(flags) + ":" + regex + ")";
058        }
059    }
060
061    /**
062     * See {@link #embedFlags(String, int)}.
063     */
064    protected String mapFlagsToEmbeddedFlags(int flags) {
065        String flagsAsString = "";
066        if ((flags & Pattern.CASE_INSENSITIVE) != 0) {
067            flagsAsString += "i";
068        }
069        if ((flags & Pattern.UNIX_LINES) != 0) {
070            flagsAsString += "d";
071        }
072        if ((flags & Pattern.MULTILINE) != 0) {
073            flagsAsString += "m";
074        }
075        if ((flags & Pattern.DOTALL) != 0) {
076            flagsAsString += "s";
077        }
078        if ((flags & Pattern.UNICODE_CASE) != 0) {
079            flagsAsString += "u";
080        }
081        if ((flags & Pattern.COMMENTS) != 0) {
082            flagsAsString += "x";
083        }
084
085        return flagsAsString;
086    }
087}
