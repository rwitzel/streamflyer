016package com.github.rwitzel.streamflyer.experimental.xml;
017
018import java.util.regex.Matcher;
019import java.util.regex.Pattern;
020
021import com.github.rwitzel.streamflyer.core.AfterModification;
022import com.github.rwitzel.streamflyer.core.Modifier;
023import com.github.rwitzel.streamflyer.experimental.stateful.State;
024import com.github.rwitzel.streamflyer.experimental.stateful.StatefulModifier;
025import com.github.rwitzel.streamflyer.experimental.stateful.util.IdleModifierState;
026import com.github.rwitzel.streamflyer.experimental.stateful.util.OneTransitionState;
027import com.github.rwitzel.streamflyer.internal.thirdparty.ZzzValidate;
028import com.github.rwitzel.streamflyer.util.ModificationFactory;
029import com.github.rwitzel.streamflyer.xml.XmlPrologRidiculouslyLongException;
030
031/**
032 * Same semantics as {@link com.github.rwitzel.streamflyer.xml.XmlVersionModifier} but the implementation uses a
033 * {@link StatefulModifier}.
034 * 
035 * @author rwoo
036 * 
037 * @since 14.09.2011
038 */
039public class XmlVersionModifier implements Modifier {
040
041    private StatefulModifier statefulModifier;
042
043    //
044    // constants
045    //
046
047    public final int INITIAL_NUMBER_OF_CHARACTERS = 4096;
048
049    //
050    // injected properties
051    //
052
053    protected ModificationFactory factory;
054
055    protected String xmlVersion;
056
057    //
058    // constructors
059    //
060
061    public XmlVersionModifier(String xmlVersion, int newNumberOfChars) {
062
063        ZzzValidate.notNull(xmlVersion, "xmlVersion must not be null");
064
065        this.factory = new ModificationFactory(0, newNumberOfChars);
066        this.xmlVersion = xmlVersion;
067
068        // (1) create the initial state: No input read yet.
069        OneTransitionState initialState = new OneTransitionState("initial") {
070
071            @Override
072            public AfterModification innerModify(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
073                    boolean endOfStreamHit) {
074
075                // you never know how many whitespace characters are in the
076                // prolog
077                return factory.modifyAgainImmediately(INITIAL_NUMBER_OF_CHARACTERS, firstModifiableCharacterInBuffer);
078            }
079        };
080
081        // (2) create version modifying state: The modifier has requested to
082        // read the XML prolog. The version of prolog is modified or a prolog is
083        // added if the prolog is missing.
084        OneTransitionState versionModifyingState = new OneTransitionState("versionModifying") {
085
086            @Override
087            public AfterModification innerModify(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
088                    boolean endOfStreamHit) {
089
090                // (Should we do aware of BOMs here? No. I consider it the
091                // responsibility of the caller to provide characters without
092                // BOM.)
093
094                Matcher matcher = Pattern.compile("<\\?xml[^>]*version\\s*=\\s*['\"]((1.0)|(1.1))['\"].*").matcher(
095                        characterBuffer);
096                if (matcher.matches()) {
097
098                    // replace version in prolog
099                    characterBuffer.replace(matcher.start(1), matcher.end(1), XmlVersionModifier.this.xmlVersion);
100                } else {
101
102                    // is there a prolog that is too long?
103                    Matcher matcher2 = Pattern.compile("<\\?xml.*").matcher(characterBuffer);
104                    if (matcher2.matches()) {
105                        // this is not normal at all -> throw exception
106                        throw new XmlPrologRidiculouslyLongException(characterBuffer.toString());
107                    }
108
109                    // insert prolog
110                    characterBuffer.insert(0, "<?xml version='" + XmlVersionModifier.this.xmlVersion + "'>");
111                }
112
113                return factory.skipEntireBuffer(characterBuffer, firstModifiableCharacterInBuffer, endOfStreamHit);
114            }
115        };
116
117        // (3) create idle state: The modifier has read the XML prolog, and has
118        // modified it if necessary. Nothing more to do for the modifier.
119        State idleState = new IdleModifierState(factory);
120
121        // (4) link the states: The state transitions are from "initial" to
122        // "version modifying" to "idle"
123        initialState.setNextState(versionModifyingState);
124        versionModifyingState.setNextState(idleState);
125
126        // 5) create the stateful modifier
127        this.statefulModifier = new StatefulModifier(initialState);
128    }
129
130    //
131    // override Modifier.*
132    //
133
134    /**
135     * @see com.github.rwitzel.streamflyer.core.Modifier#modify(java.lang.StringBuilder, int, boolean)
136     */
137    @Override
138    public AfterModification modify(StringBuilder characterBuffer, int firstModifiableCharacterInBuffer,
139            boolean endOfStreamHit) {
140
141        // delegate
142        return statefulModifier.modify(characterBuffer, firstModifiableCharacterInBuffer, endOfStreamHit);
143    }
144}
