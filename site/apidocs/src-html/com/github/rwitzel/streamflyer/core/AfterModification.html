016package com.github.rwitzel.streamflyer.core;
017
018import com.github.rwitzel.streamflyer.internal.thirdparty.ZzzValidate;
019
020/**
021 * This class is a message from the modifier to the stream processor, or technically spoken, this class represents the
022 * return value of {@link Modifier#modify(StringBuilder, int, boolean)}. This return value defines how the stream
023 * processor, i. e. either a {@link ModifyingReader} or a {@link ModifyingWriter}, shall behave before calling
024 * {@link Modifier#modify(StringBuilder, int, boolean)} again.
025 * <p>
026 * Therefore, this class defines
027 * <ul>
028 * <li>how many {@link AfterModification#getNewNumberOfChars() modifiable characters} the stream processor shall add to
029 * the character buffer,
030 * <li>how many {@link AfterModification#getNewMinimumLengthOfLookBehind() unmodifiable characters} the stream processor
031 * shall keep in the character buffer,
032 * <li>{@link AfterModification#isModifyAgainImmediately() whether} modifiable characters shall be shall be made
033 * unmodifiable, and if so, how many {@link AfterModification#getNewMinimumLengthOfLookBehind() modifiable characters}
034 * shall be made unmodifiable.
035 * </ul>
036 * <p>
037 * If you want to implement your own modifier, then you should distinct three different types of
038 * {@link AfterModification} messages - depending on how many characters shall be skipped and how many characters are
039 * left in the buffer and how many characters are left in the entire stream:
040 * <ul>
041 * <li>SKIP: Skip at least one modifiable character in the buffer, or do not skip any character if there are no
042 * modifiable characters are left in the buffer but the end of stream is not hit yet. A synonym for SKIP could be
043 * CONTINUE.
044 * <li>MODIFY AGAIN IMMEDIATELY: Do not skip any character. This type is useful only if the end of stream is not hit yet
045 * and the modifier needs more {@link #getNewNumberOfChars() input} to do its job.
046 * <li>STOP: Do not skip any character. This requires that the end of stream is hit and no modifiable characters are
047 * left in the buffer.
048 * <li>(FETCH_MORE_INPUT: This is a synonym for SKIP if some characters can be skipped. Otherwise it is a MODIFY AGAIN
049 * IMMEDIATELY.)
050 * </ul>
051 * 
052 * @author rwoo
053 * @since 03.06.2011
054 */
055public class AfterModification {
056
057    //
058    // injected
059    //
060
061    /**
062     * See <code>ModifyingReader#newMinimumLengthOfLookBehind</code>.
063     * <p>
064     * The value of this property must be a non-negative number.
065     */
066    private int newMinimumLengthOfLookBehind;
067
068    /**
069     * See {@link #getNewNumberOfChars()}
070     */
071    private int newNumberOfChars;
072
073    /**
074     * See {@link #getNumberOfCharactersToSkip()}
075     */
076    private int numberOfCharactersToSkip;
077
078    /**
079     * See {@link #isModifyAgainImmediately()}.
080     */
081    private boolean modifyAgainImmediately;
082
083    //
084    // constructors
085    //
086
087    /**
088     * @param numberOfCharactersToSkip
089     *            See {@link #getNumberOfCharactersToSkip()}
090     * @param modifyAgainImmediately
091     *            See {@link #isModifyAgainImmediately()}
092     * @param newMinimumLengthOfLookBehind
093     *            See {@link #getNewMinimumLengthOfLookBehind()}
094     * @param newNumberOfChars
095     *            See {@link #getNewNumberOfChars()}
096     */
097    public AfterModification(int numberOfCharactersToSkip, boolean modifyAgainImmediately,
098            int newMinimumLengthOfLookBehind, int newNumberOfChars) {
099
100        init(numberOfCharactersToSkip, modifyAgainImmediately, newMinimumLengthOfLookBehind, newNumberOfChars);
101
102    }
103
104    protected void init(int numberOfCharactersToSkip_, boolean modifyAgainImmediately_,
105            int newMinimumLengthOfLookBehind_, int newNumberOfChars_) {
106
107        ZzzValidate.isZeroOrPositiveNumber(numberOfCharactersToSkip_, "numberOfCharactersToSkip");
108        ZzzValidate.isZeroOrPositiveNumber(newMinimumLengthOfLookBehind_, "minimumLengthOfLookBehind");
109        // (newNumberOfChars should not be zero if end of stream is
110        // not hit but we cannot check this constraint here as in this method we
111        // don't know whether the end of stream is hit.)
112        ZzzValidate.isZeroOrPositiveNumber(newNumberOfChars_, "newNumberOfChars_");
113
114        if (modifyAgainImmediately && numberOfCharactersToSkip_ != 0) {
115            throw new IllegalArgumentException("if modify again immediately,"
116                    + " the number of characters to skip are ignored .. "
117                    + "so why is the number of characters to skip " + numberOfCharactersToSkip_ + " instead of zero?");
118        }
119
120        this.numberOfCharactersToSkip = numberOfCharactersToSkip_;
121        this.modifyAgainImmediately = modifyAgainImmediately_;
122        this.newMinimumLengthOfLookBehind = newMinimumLengthOfLookBehind_;
123        this.newNumberOfChars = newNumberOfChars_;
124    }
125
126    //
127    // getter
128    //
129
130    /**
131     * @return Returns the {@link #newMinimumLengthOfLookBehind}.
132     *         <p>
133     *         The value must be a positive number or zero.
134     */
135    public int getNewMinimumLengthOfLookBehind() {
136        return newMinimumLengthOfLookBehind;
137    }
138
139    /**
140     * @return Returns the number of unmodifiable characters (look-behind) and modifiable characters in the buffer that
141     *         shall be passed to {@link Modifier#modify(StringBuilder, int, boolean)} the next time this method is
142     *         called.
143     *         <p>
144     *         Unless the end of the stream is hit, {@link ModifyingReader} and {@link ModifyingWriter} do not provide
145     *         fewer characters to the modifier than requested by the value returned here.
146     *         <p>
147     *         The value must be a positive number greater than zero.
148     */
149    public int getNewNumberOfChars() {
150        return newNumberOfChars;
151    }
152
153    /**
154     * @return Returns true if the modifier shall be called again immediately.
155     *         <p>
156     *         This is usually true either if we want to edit the insertion itself or if the modifier has to read more
157     *         characters to decide again about an appropriate modification. The latter one is particularly useful for
158     *         modifiers that use (greedy) regular expressions.
159     *         <p>
160     *         See also {@link #getNumberOfCharactersToSkip()}.
161     */
162    public boolean isModifyAgainImmediately() {
163        return modifyAgainImmediately;
164    }
165
166    /**
167     * @return Returns the number of characters to skip if the modification shall no be tried again, i.e. if
168     *         {@link #isModifyAgainImmediately()} is false. In this case this number must be at least one unless the
169     *         end of the stream is hit.
170     *         <p>
171     *         After the modifiable characters are investigated by the modifier and some characters might be deleted or
172     *         inserted, you have to specify the number of characters that are no longer considered modifiable. If the
173     *         end of the stream is not hit and {@link #isModifyAgainImmediately()} returns false, then this method must
174     *         return at least 1 -- so that the processing of the characters of the stream makes any progress.
175     *         <p>
176     *         In all cases the value of this property must be a non-negative number.
177     *         <p>
178     *         This property will be ignored by {@link ModifyingReader} and {@link ModifyingWriter} if
179     *         {@link #isModifyAgainImmediately()} returns true.
180     */
181    public int getNumberOfCharactersToSkip() {
182        return numberOfCharactersToSkip;
183    }
184
185    /**
186     * @return Returns an informal description of the type of the message, i.e. either SKIP or MODIFY AGAIN IMMEDIATELY
187     *         or STOP.
188     */
189    public String getMessageType() {
190        if (numberOfCharactersToSkip != 0) {
191            return "SKIP";
192        } else if (modifyAgainImmediately) {
193            return "MODIFY AGAIN IMMEDIATELY";
194        } else {
195            return "STOP";
196        }
197    }
198
199    //
200    // Object.*
201    //
202
203    /**
204     * @see java.lang.Object#toString()
205     */
206    @Override
207    public String toString() {
208        StringBuilder builder = new StringBuilder();
209        builder.append("AfterModification [\nnewMinimumLengthOfLookBehind=");
210        builder.append(newMinimumLengthOfLookBehind);
211        builder.append(", \nnewNumberOfModifiableChars=");
212        builder.append(newNumberOfChars);
213        builder.append(", \nnumberOfCharactersToSkip=");
214        builder.append(numberOfCharactersToSkip);
215        builder.append(", \nmodifyAgainImmediately=");
216        builder.append(modifyAgainImmediately);
217        builder.append("]");
218        return builder.toString();
219    }
220}
