018import java.io.BufferedWriter;
019import java.io.IOException;
020import java.io.Writer;
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;

027/**
028 * A {@link Writer} that allows a {@link Modifier} to modify the characters in the underlying stream.
029 * <p>
030 * This class is not synchronized.
031 * <p>
032 * ATTENTION! This writer flushes only characters that are confirmed by the given {@link Modifier}. EXAMPLE: Assume you
033 * wrote 50 bytes to the writer and the modifier confirmed 25 of them by
034 * {@link AfterModification#getNumberOfCharactersToSkip() skipping } them, then subsequent flushing will write only
035 * these 25 confirmed characters.
036 * 
037 * @author rwoo
038 * @since 06.05.2011
039 */
040public class ModifyingWriter extends Writer {
041
042    //
043    // injected properties
044    //
045
046    protected Writer delegate;
047
048    /**
049     * The modifier thats replaces, deletes, inserts characters of the underlying stream, and manages the buffer size.
050     */
051    private Modifier modifier;
052
053    //
054    // properties that represent the mutable state
055    //
056
057    /**
058     * At its begin the buffer contains unmodifiable characters (might be needed for look-behind matches as known from
059     * regular expression matching). After the unmodifiable characters the buffer holds the modifiable characters. See
060     * {@link #firstModifiableCharacterInBuffer}.
061     */
062    private StringBuilder characterBuffer;
063
064    /**
065     * The position of the first character in the {@link #characterBuffer input buffer} that is modifiable.
066     * <p>
067     * This character and the following characters in {@link #characterBuffer} can be modified by the {@link #modifier}.
068     */
069    private int firstModifiableCharacterInBuffer;
070
071    /**
072     * The value is taken from {@link AfterModification#getNewMinimumLengthOfLookBehind()} after
073     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. The value is <em>requested</em> by the
074     * {@link Modifier}.
075     * <p>
076     * If this value is greater than {@link #firstModifiableCharacterInBuffer}, the modifier is probably faulty.
077     */
078    private int minimumLengthOfLookBehind;
079
080    /**
081     * The value is the sum of {@link AfterModification#getNewMinimumLengthOfLookBehind()} and
082     * {@link AfterModification#getNewNumberOfChars()} after {@link Modifier#modify(StringBuilder, int, boolean)} is
083     * called. The value is <em>requested</em> by the {@link Modifier}.
084     */
085    private int requestedNumCharactersInBuffer;
086
087    /**
088     * The value is initially taken from {@link AfterModification#getNumberOfCharactersToSkip()} after
089     * {@link Modifier#modify(StringBuilder, int, boolean)} is called. Then during reading characters from the reader
090     * this value is decremented.
091     */
092    private int numberOfCharactersToSkip;
093
094    /**
095     * True if the end of stream was detected.
096     */
097    private boolean endOfStreamHit = false;
098
099    /**
100     * The holds the last modification provided by the {@link #modifier}. This property serves debugging purposes only.
101     */
102    private AfterModification lastAfterModificationForDebuggingOnly = null;
103
104    /**
105     * @param writer
106     *            The underlying writer that provides the original, not modified characters. For optimal performance
107     *            choose a {@link BufferedWriter}.
108     * @param modifier
109     */
110    public ModifyingWriter(Writer writer, Modifier modifier) {
111        super(writer);
112
113        // take injected properties
114        this.modifier = modifier;
115        this.delegate = writer;
116
117        // initialize the mutable state
118        this.numberOfCharactersToSkip = 0;
119        this.minimumLengthOfLookBehind = 0; // we cannot do anything else
120        this.firstModifiableCharacterInBuffer = 0;
121        this.requestedNumCharactersInBuffer = 1; // any value
122        this.characterBuffer = new StringBuilder(requestedNumCharactersInBuffer);
123        // adjustCapacityOfBuffer();
124    }
125
126    //
127    // interface Writer
128    //
129
130    /**
131     * @see java.io.Writer#flush()
132     */
133    @Override
134    public void flush() throws IOException {
135
136        // delegate to the underlying writer
137        delegate.flush();
138    }
139
140    /**
141     * Flushes everything. Closes the underlying stream.
142     * 
143     * @see java.io.Writer#close()
144     */
145    @Override
146    public void close() throws IOException {
147        close(true);
148    }
149
150    /**
151     * Similar to {@link #close()} but the caller can choose whether or not to close the underlying writer.
152     * 
153     * @param closeUnderlyingWriter
154     *            True if the underlying writer shall be closed as well.
155     * @throws IOException
156     */
157    public void close(boolean closeUnderlyingWriter) throws IOException {
158
159        if (endOfStreamHit) {
160            return; // nothing to
161        }
162
163        endOfStreamHit = true;
164
165        // modify as long as characters are skipped (modify again makes not
166        // much sense but there may some use cases that require modfiyAgain even
167        // on the position after the last character)
168        while (modify()) {
169            // loop
170        }
171
172        // write the remaining bytes to the underlying writer
173        delegate.append(characterBuffer, firstModifiableCharacterInBuffer, characterBuffer.length());
174
175        // clear same variables (only to tidy up)
176        characterBuffer = null;
177
178        flush();
179
180        if (closeUnderlyingWriter) {
181            delegate.close();
182        }
183    }
184
185    /**
186     * @see java.io.Writer#write(char[], int, int)
187     * @param off
188     *            the first character in cbuf not appended to characterBuffer yet
189     * @param len
190     *            the number of characters in cbuf not appended to // characterBuffer yet
191     */
192    @Override
193    public void write(char[] cbuf, int off, int len) throws IOException {
194
195        // is the stream already closed?
196        if (endOfStreamHit) {
197            throw new IOException("the stream is already closed");
198        }
199
200        // are there some characters to append to the characterBuffer yet?
201        while (len > 0) {
202            // yes -> append characters
203
204            // determine the number of characters to add to the buffer
205            int numberOfCharactersToAppend;
206            // are there are more characters available than requested?
207            if (len > requestedNumCharactersInBuffer - characterBuffer.length()) {
208                // yes -> we won't append more characters than are requested (in
209                // order to save memory)
210                numberOfCharactersToAppend = requestedNumCharactersInBuffer - characterBuffer.length();
211            } else {
212                // we will append all given characters
213                numberOfCharactersToAppend = len;
214            }
215
216            // append the characters to the characterBuffer
217            characterBuffer.append(cbuf, off, numberOfCharactersToAppend);
218
219            // update the variables offset and length
220            off = off + numberOfCharactersToAppend;
221            len = len - numberOfCharactersToAppend;
222
223            // modify if there are enough characters in the buffer
224            while (characterBuffer.length() >= requestedNumCharactersInBuffer && modify()) {
225                // loop
226            }
227
228        }
229    }
230
231    /**
232     * @return Returns true if some characters are skipped, i.e. written to the underlying writer, or the content of the
233     *         buffer shall be modified again immediately.
234     * @throws IOException
235     */
236    private boolean modify() throws IOException {
237        AfterModification afterModification = modifier.modify(characterBuffer, firstModifiableCharacterInBuffer,
238                endOfStreamHit);
239
240        lastAfterModificationForDebuggingOnly = afterModification;
241
242        numberOfCharactersToSkip = afterModification.getNumberOfCharactersToSkip();
243
244        boolean someCharactersSkippedOrModifyAgainImmediately = false;
245
246        if (afterModification.isModifyAgainImmediately()) {
247
248            if (endOfStreamHit) {
249                // A modifier might do "modify again" although there is no
250                // more input. Assume the modifier want to edit its own
251                // insertions -> Therefore, we throw no exception here.
252            }
253
254            // nothing to do apart from adjusting capacity and look-behind
255
256            someCharactersSkippedOrModifyAgainImmediately = true;
257
258        } else {
259
260            // write away the locked characters
261            if (numberOfCharactersToSkip > 0) {
262                int end = firstModifiableCharacterInBuffer + numberOfCharactersToSkip;
263                if (end > characterBuffer.length()) {
264                    onFaultyModifier(-51, String.format("You try to skip " + "characters that you have not"
265                            + " seen yet(%s %s %s %s)", firstModifiableCharacterInBuffer, numberOfCharactersToSkip,
266                            characterBuffer.length(), characterBuffer));
267                }
268
269                delegate.append(characterBuffer, firstModifiableCharacterInBuffer, end);
270                someCharactersSkippedOrModifyAgainImmediately = true;
271                firstModifiableCharacterInBuffer = end;
272            } else { // if (numberOfCharactersToSkip == 0)
273
274                // This block is usually entered when the modifier decided to do
275                // nothing at the position after the last characters of the
276                // stream
277
278            }
279        }
280
281        // update minimumLengthOfLookBehind
282        minimumLengthOfLookBehind = afterModification.getNewMinimumLengthOfLookBehind();
283        if (minimumLengthOfLookBehind > firstModifiableCharacterInBuffer) {
284            onFaultyModifier(-11, "Requested Look Behind" + " is impossible because there are not enough "
285                    + "characters in the stream.");
286        }
287
288        // update requestedNumCharactersInBuffer
289        requestedNumCharactersInBuffer = minimumLengthOfLookBehind + afterModification.getNewNumberOfChars();
290        if (requestedNumCharactersInBuffer < minimumLengthOfLookBehind + 1) {
291            onFaultyModifier(-13, "Requested Capacity" + " is two small because there must at least one"
292                    + " unread characters available after the " + "look behind characters characters in the"
293                    + " stream.");
294        }
295
296        updateBuffer();
297
298        return someCharactersSkippedOrModifyAgainImmediately;
299    }
300
301    /**
302     * Updates the input buffer according to {@link #minimumLengthOfLookBehind} and
303     * {@link #requestedNumCharactersInBuffer}, and then fills the buffer up to its capacity.
304     */
305    private void updateBuffer() {
306
307        removeCharactersInBufferNotNeededAnyLonger();
308
309        adjustCapacityOfBuffer();
310
311    }
312
313    /**
314     * Deletes those characters at the start of the buffer we do not need any longer.
315     */
316    private void removeCharactersInBufferNotNeededAnyLonger() {
317
318        int charactersToDelete = firstModifiableCharacterInBuffer - minimumLengthOfLookBehind;
319
320        if (charactersToDelete > 0) {
321            characterBuffer.delete(0, charactersToDelete);
322            firstModifiableCharacterInBuffer -= charactersToDelete;
323        } else if (charactersToDelete < 0) {
324            onFaultyModifier(-52, charactersToDelete + " characters to delete but this is not possible ("
325                    + firstModifiableCharacterInBuffer + "," + minimumLengthOfLookBehind + ")");
326        }
327    }
328
329    /**
330     * Adjusts the capacity of the buffer.
331     */
332    private void adjustCapacityOfBuffer() {
333
334        // is the current capacity not big enough?
335        if (characterBuffer.capacity() < requestedNumCharactersInBuffer) {
336
337            // increase the capacity (we delegate to the default behavior of
338            // StringBuffer)
339            characterBuffer.ensureCapacity(requestedNumCharactersInBuffer);
340
341        }
342
343    }
344
345    protected void onFaultyModifier(int errorCode, String errorMessage) {
346        // should we silently ignore any errors and fallback to a meaningful
347        // behavior? No because a faulty modifier should be fixed by a
348        // ValidModifier, i.e. the user must take the responsibility for the
349        // errors and deal with them. Therefore, we throw an exception here.
350
351        // (1) collect the current state of the involved objects
352        Map<String, Object> description = new HashMap<String, Object>();
353
354        // describe arguments
355        description.put("argument errorCode", errorCode);
356        description.put("argument errorMessage", errorMessage);
357
358        // describe reader
359        description.put("this", this);
360
361        // (2) create an error message
362        List<String> keys = new ArrayList<String>(description.keySet());
363        Collections.sort(keys);
364        StringBuilder sb = new StringBuilder();
365        for (String key : keys) {
366            sb.append("\n" + key + ": " + description.get(key));
367        }
368
369        // (3) throw exception
370        throw new FaultyModifierException(sb.toString(), description);
371    }

375     */
376    @Override
377    public String toString() {
378        StringBuilder builder = new StringBuilder();
379        builder.append("ModifyingWriter [\ndelegate=");
380        builder.append(delegate);
381        builder.append(", \nmodifier=");
382        builder.append(modifier);
383        builder.append(", \ncharacterBuffer=");
384        builder.append(characterBuffer);
385        builder.append(", \nnextUnreadCharacterInBuffer=");
386        builder.append(firstModifiableCharacterInBuffer);
387        builder.append(", \nminimumLengthOfLookBehind=");
388        builder.append(minimumLengthOfLookBehind);
389        builder.append(", \nrequestedCapacityOfCharacterBuffer=");
390        builder.append(requestedNumCharactersInBuffer);
391        builder.append(", \nlockedCharacters=");
392        builder.append(numberOfCharactersToSkip);
393        builder.append(", \nendOfStreamHit=");
394        builder.append(endOfStreamHit);
395        builder.append(", \nlastModificationForDebuggingOnly=");
396        builder.append(lastAfterModificationForDebuggingOnly);
397        builder.append("]");
398        return builder.toString();
399    }
